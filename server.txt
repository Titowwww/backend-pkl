// Helper function to upload files to Firebase
const uploadFileToFirebase = async (buffer, filename, mimetype) => {
  const blob = bucket.file(`${uuidv4()}_${filename}`);
  const blobStream = blob.createWriteStream({
    metadata: {
      contentType: mimetype,
    },
  });

  return new Promise((resolve, reject) => {
    blobStream.on('error', (err) => {
      reject(err);
    });

    blobStream.on('finish', async () => {
      const publicUrl = `https://storage.googleapis.com/${bucket.name}/${blob.name}`;
      resolve(publicUrl);
    });

    blobStream.end(buffer);
  });
};

// Middleware to handle base64 file uploads
const handleBase64Files = (req, res, next) => {
  const base64Fields = ['suratPengantarFile', 'proposalFile', 'ktpFile'];

  base64Fields.forEach(field => {
    if (req.body[field]) {
      const base64Data = req.body[field];
      const matches = base64Data.match(/^data:(.+);base64,(.+)$/);
      if (matches) {
        const mimetype = matches[1];
        const base64File = matches[2];
        req.body[`${field}Buffer`] = Buffer.from(base64File, 'base64');
        req.body[`${field}Mimetype`] = mimetype;
        req.body[`${field}Name`] = `file_${uuidv4()}`; // Generate a unique filename or use an existing filename
      }
    }
  });
  next();
};

// Endpoint to handle research form submission
app.post('/api/penelitian', handleBase64Files, validateResearchForm, async (req, res) => {
  const {
    letterNumber,
    name,
    researcherName,
    address,
    inputValue,
    institution,
    occupation,
    judulPenelitian,
    researchField,
    tujuanPenelitian,
    supervisorName,
    teamMembers,
    statusPenelitian,
    researchPeriod,
    researchLocation,
    suratPengantarFileBuffer,
    suratPengantarFileName,
    suratPengantarFileMimetype,
    proposalFileBuffer,
    proposalFileName,
    proposalFileMimetype,
    ktpFileBuffer,
    ktpFileName,
    ktpFileMimetype
  } = req.body;

  try {
    // Log data received in the request
    console.log("Request Body:", req.body);

    // Upload files to Firebase Storage
    const suratPengantarUrl = suratPengantarFileBuffer ? await uploadFileToFirebase(suratPengantarFileBuffer, suratPengantarFileName, suratPengantarFileMimetype) : null;
    const proposalUrl = proposalFileBuffer ? await uploadFileToFirebase(proposalFileBuffer, proposalFileName, proposalFileMimetype) : null;
    const ktpUrl = ktpFileBuffer ? await uploadFileToFirebase(ktpFileBuffer, ktpFileName, ktpFileMimetype) : null;

    // Save data to Firestore
    await db.collection('pelayanan').doc('penelitian').collection('data').add({
      letterNumber,
      name,
      researcherName,
      address,
      inputValue,
      institution,
      occupation,
      judulPenelitian,
      researchField,
      tujuanPenelitian,
      supervisorName,
      teamMembers,
      statusPenelitian,
      researchPeriod,
      researchLocation,
      suratPengantarUrl,
      proposalUrl,
      ktpUrl,
      timestamp: admin.firestore.FieldValue.serverTimestamp()
    });

    res.json({ message: 'Data berhasil disimpan' });
  } catch (err) {
    console.error(err);
    res.status(500).json({ message: 'Terjadi kesalahan saat menyimpan data' });
  }
});
